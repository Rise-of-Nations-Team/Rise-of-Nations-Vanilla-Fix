						all_provinces = yes
					}
					building = supply_node
					level > 0
				}
			}
		}
		value = -5
	}

	# Prioritize capital area
	ai_strategy = {
		type = front_unit_request
		state_trigger = {
			OR = {
				is_capital = yes
				any_neighbor_state = { is_capital = yes }
			}
		}
		value = 15
	}

	# Heavily prioritize own capital
	ai_strategy = {
		type = front_unit_request
		state_trigger = {
			is_capital = yes
			owner = { tag = FROM.FROM }
		}
		value = 50
	}

	# Reinforce supply hub corridors
	ai_strategy = {
		type = front_unit_request
		state_trigger = {
			is_controlled_by = FROM.FROM
			any_neighbor_state = {
				any_province_building_level = {
					province = { all_provinces = yes }
					building = supply_node
					level > 0
				}
			}
		}
		value = 10
	}

	# Deprioritize low-value naval invasions
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			NOT = { has_state_flag = recent_naval_invasion_flag }
			check_variable = { state_strategic_value < 3 }
		}
		value = -5
	}

	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			NOT = { is_core_of = FROM.FROM }
			NOT = { has_state_flag = recent_naval_invasion_flag }
			check_variable = { state_strategic_value < 8 }
		}
		value = -3

invasion_prefer_ports_and_real_targets = {
	enable = {
		has_war = yes
		has_crisis = no
	}

	abort_when_not_enabled = yes

	# Ports are the main compatible target. They create supply and stop landings from becoming trapped beach raids.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			any_province_building_level = {
				province = { all_provinces = yes }
				building = naval_base
				level > 0
			}
		}
		value = 35
	}

	# High-value enemy-controlled states stay attractive even in overhaul maps.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			check_variable = { state_strategic_value > 15 }
		}
		value = 20
	}

	# Enemy cores usually matter more than random occupied coastline.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			is_core_of = FROM
		}
		value = 12
	}

	# Port targets next to our land are excellent follow-up targets: short path, immediate front pressure.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			any_neighbor_state = {
				is_controlled_by = FROM.FROM
			}
			any_province_building_level = {
				province = { all_provinces = yes }
				building = naval_base
				level > 0
			}
		}
		value = 25
	}

	# Random non-port beaches are usually how the AI throws divisions into supply death.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			NOT = {
				any_province_building_level = {
					province = { all_provinces = yes }
					building = naval_base
					level > 0
				}
			}
		}
		value = -30
	}

	# Low-value non-port beaches get buried hard unless another strategy has a very strong reason.
	ai_strategy = {
		type = invasion_unit_request
		state_trigger = {
			controller = {
				has_war_with = FROM.FROM
			}
			check_variable = { state_strategic_value < 8 }
			NOT = {
				any_province_building_level = {
					province = { all_provinces = yes }
					building = naval_base
					level > 0
				}
			}
		}
		value = -60
	}
}
